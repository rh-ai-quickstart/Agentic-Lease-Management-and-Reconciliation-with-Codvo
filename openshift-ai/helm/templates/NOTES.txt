================================================================================
  QUICK START
================================================================================

1. Verify all pods are running:
   oc get pods -n {{ .Values.global.namespace }}

2. Check the logs:
{{- if .Values.llmd.enabled }}
   oc logs -f deployment/{{ .Values.llmd.name }} -n {{ .Values.global.namespace }}
{{- end }}
{{- if .Values.vllm.enabled }}
   oc logs -f deployment/{{ .Values.vllm.model.name }}-predictor -n {{ .Values.global.namespace }}
{{- end }}

3. Test the inference endpoint:
{{- if .Values.vllm.enabled }}
   curl -X POST "https://$ROUTE_HOST/v1/chat/completions" \
     -H "Content-Type: application/json" \
     -d '{
       "model": "{{ .Values.vllm.model.path }}",
       "messages": [{"role": "user", "content": "Hello, how can you help with lease management?"}],
       "max_tokens": 100
     }'
{{- end }}
